测试功能 - 编程特攻队

测试运算符

#include using namespace std; int main() { int a = 10, b = 3; cout << "a + b = " << a + b << endl; cout << "a - b = " << a - b << endl; cout << "a * b = " << a * b << endl; cout << "a / b = " << a / b << endl; cout << "a % b = " << a % b << endl; bool x = true, y = false; cout << "x && y: " << (x && y) << endl; cout << "x || y: " << (x || y) << endl; cout << "!x: " << !x << endl; return 0; }

测试转义符

#include using namespace std; int main() { cout << "换行符: line1\nline2" << endl; cout << "制表符: col1\tcol2\tcol3" << endl; cout << "单引号: \'Hello\'" << endl; cout << "双引号: \"World\"" << endl; cout << "反斜杠: \\path\\to\\file" << endl; return 0; }

测试二进制编码

#include using namespace std; int main() { int binary1 = 0b1010; // 10 int binary2 = 0b1111; // 15 int binary3 = 0b10000000; // 128 cout << "0b1010 = " << binary1 << endl; cout << "0b1111 = " << binary2 << endl; cout << "0b10000000 = " << binary3 << endl; return 0; }

测试算法

#include #include using namespace std; int main() { cout << "max(10, 20) = " << max(10, 20) << endl; cout << "min(10, 20) = " << min(10, 20) << endl; cout << "sqrt(16) = " << sqrt(16) << endl; cout << "abs(-5) = " << abs(-5) << endl; cout << "pow(2, 3) = " << pow(2, 3) << endl; return 0; }

测试数据结构

#include #include using namespace std; int main() { vector vec = {1, 2, 3, 4, 5}; cout << "Vector 元素: "; for (int v : vec) { cout << v << " "; } cout << endl; return 0; }
