3. 循环和条件判断

using namespace std;

    cout << "Even numbers from 1 to 20:" << endl;
    for (int i = 1; i <= 20; i++) {
        if (i % 2 == 0) {
            cout << i << " ";
        }
    }
    cout << endl;
    
    return 0;
}
4. 数组操作
#include <iostream>
using namespace std;

int main() {
    int numbers[] = {10, 20, 30, 40, 50};
    int sum = 0;
    
    cout << "Array elements: ";
    for (int i = 0; i < 5; i++) {
        cout << numbers[i] << " ";
        sum += numbers[i];
    }
    cout << endl;
    
    cout << "Sum: " << sum << endl;
    cout << "Average: " << static_cast<double>(sum) / 5 << endl;
    
    return 0;
}
5. 函数调用
#include <iostream>
using namespace std;

int factorial(int n) {
    if (n == 0 || n == 1) {
        return 1;
    }
    return n * factorial(n - 1);
}

int main() {
    for (int i = 1; i <= 5; i++) {
        cout << i << "! = " << factorial(i) << endl;
    }
    return 0;
}
6. 输入输出
#include <iostream>

    cout << "Sum: " << a + b << endl;
    cout << "Difference: " << a - b << endl;
    cout << "Product: " << a * b << endl;
    
    if (b != 0) {
        cout << "Quotient: " << static_cast<double>(a) / b << endl;

        cout << "Cannot divide by zero!" << endl;
    }
    
    return 0;
}
7. 二维数组（矩阵）
#include <iostream>
using namespace std;

int main() {
    int matrix[3][3] = {
        {1, 2, 3},
        {4, 5, 6},
        {7, 8, 9}
    };
    
    cout << "3x3 Matrix:" << endl;
    for (int i = 0; i < 3; i++) {
        for (int j = 0; j < 3; j++) {
            cout << matrix[i][j] << " ";
        }
        cout << endl;
    }
    
    return 0;
}
