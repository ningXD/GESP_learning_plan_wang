💻 常用代码模板

1. 数位分离模板（求数位和、判断数位特征）

int n, sum = 0;
cin >> n;
while (n > 0) {
    sum += n % 10;  // 取个位数
    n /= 10;        // 去掉个位数
}

2. 质数判断模板（优化版）

bool isPrime(int n) {
    if (n < 2) return false;
    for (int i = 2; i * i <= n; i++) {  // 优化：只需检查到sqrt(n)
        if (n % i == 0) return false;
    }
    return true;
}

3. 矩阵绘制通用模板

int n;
cin >> n;
for (int i = 0; i < n; i++) {
    for (int j = 0; j < n; j++) {
        if (/* 满足某个条件 */) {
            cout << "+";
        } else {
            cout << "-";
        }
    }
    cout << endl;
}
